   \begin{equation}
     \sum_{j=2}^n \frac{1}{j} < \int_1^n \frac{1}{x} \,dx < \sum_{j = 1}^{n-1} \frac{1}{j} < \sum_{j = 1}^{n} \frac{1}{j}\\

     \int_1^n \frac{1}{x} \,dx = \log{x} \vert_1^n  = \log{n} - \log{1} = \log{n}\\

     H(n) - 1 \leq \log{n} \leq H(n)
   \end{equation}
